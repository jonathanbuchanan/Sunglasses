157  void generate() {
158  if (type == SunTextureTypeNoiseHemisphere) {
159  vector<glm::vec3> noise;
160  for (int i = 0; i < width * height; i++) {
161  glm::vec3 value = glm::vec3(randomFloats(generator) * 2.0 - 1.0, randomFloats(generator) * 2.0 - 1.0, 0.0);
162  noise.push_back(value);
163  }
164 
165  glGenTextures(1, &id);
166  glBindTexture(GL_TEXTURE_2D, id);
167  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB16F, width, height, 0, GL_RGB, GL_FLOAT, &noise[0]);
168  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
169  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
170  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
171  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
172  }
173  }
